**로컬 브랜치 생성 및 업로드**

```bash
# 1. 기준 브랜치(main)로 이동
git switch main

# 2. 깃허브의 최신 코드를 로컬 main에 반영
git pull origin main

# 3. 새 브랜치 생성 및 이동
git switch -c feat/my-feature

# 4. (코드 작업...)
# ...

# 5. 변경된 파일들을 스테이징
git add .

# 6. 커밋 메시지와 함께 변경 사항을 기록
git commit -m "feat: 나의 멋진 기능"

# 7. 깃허브에 업로드 및 로컬-원격 연결
git push -u origin feat/my-feature
```
